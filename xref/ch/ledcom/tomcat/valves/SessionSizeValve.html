16  package ch.ledcom.tomcat.valves;
17  
18  import static java.lang.String.format;
19  
20  import java.io.IOException;
21  import java.io.ObjectOutputStream;
22  import java.util.Enumeration;
23  
24  import javax.annotation.Nullable;
25  import javax.servlet.ServletException;
26  import javax.servlet.http.HttpSession;
27  
28  import org.apache.catalina.connector.Request;
29  import org.apache.catalina.connector.Response;
30  import org.apache.catalina.valves.ValveBase;
31  import org.apache.juli.logging.Log;
32  import org.apache.juli.logging.LogFactory;
33  
34  import com.google.common.io.ByteStreams;
35  import com.google.common.io.Closer;
36  import com.google.common.io.CountingOutputStream;
37  
38  /**
39   * Logs the size of the HTTP session.
40   *
41   * For this valve to work, a Java Agent must be registered to initialize the
42   * object-explorer introspection. See README.md for details.
43   *
44   * @author gehel
45   */
46  public class SessionSizeValve extends ValveBase {
47  
48      /** logger. */
49      private static Log log = LogFactory.getLog(SessionSizeValve.class);
50  
51      /**
52       * Logs the size of the HTTP session.
53       *
54       * @param request
55       *            the request being served
56       * @param response
57       *            the response being generated
58       * @throws IOException
59       * @throws ServletException
60       */
61      @Override
62      public final void invoke(final Request request, final Response response)
63              throws IOException, ServletException {
64          try {
65              getNext().invoke(request, response);
66          } finally {
67              try {
68                  final long sessionSize = measureSerializedSessionSize(request
69                          .getSession(false));
70                  log.info(format("Session size = %d.", sessionSize));
71              } catch (final IOException ioe) {
72                  log.warn("Problem measuring session size", ioe);
73              }
74          }
75      }
76  
77      private long measureSerializedSessionSize(
78              @Nullable final HttpSession session) throws IOException {
79          if (session == null) {
80              return 0;
81          }
82          @SuppressWarnings("unchecked")
83          final Enumeration<String> attibuteNames = session.getAttributeNames();
84          long sessionSize = 0;
85          while (attibuteNames.hasMoreElements()) {
86              final String attributeName = attibuteNames.nextElement();
87              sessionSize += measureSerializedSize(session
88                      .getAttribute(attributeName));
89          }
90          return sessionSize;
91      }
92  
93      private long measureSerializedSize(final Object attribute)
94              throws IOException {
95          final Closer closer = Closer.create();
96          try {
97              final CountingOutputStream countingStream = closer
98                      .register(new CountingOutputStream(ByteStreams
99                              .nullOutputStream()));
100             final ObjectOutputStream out = closer
101                     .register(new ObjectOutputStream(countingStream));
102             out.writeObject(attribute);
103             return countingStream.getCount();
104         } finally {
105             closer.close();
106         }
107     }
108 }
