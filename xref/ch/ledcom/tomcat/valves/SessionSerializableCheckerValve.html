14  package ch.ledcom.tomcat.valves;

16  import static java.lang.String.format;
17  
18  import java.io.IOException;
19  import java.io.ObjectOutputStream;
20  import java.io.Serializable;
21  import java.util.Enumeration;
22  
23  import javax.servlet.ServletException;
24  
25  import org.apache.catalina.connector.Request;
26  import org.apache.catalina.connector.Response;
27  import org.apache.catalina.valves.ValveBase;
28  import org.apache.juli.logging.Log;
29  import org.apache.juli.logging.LogFactory;
30  
31  import com.google.common.io.ByteStreams;
32  import com.google.common.io.Closer;
33  
34  /**
35   * Checks if all session attributes are serializable.
36   *
37   * @author gehel
38   */
39  public class SessionSerializableCheckerValve extends ValveBase {
40  
41      /** logger. */
42      private static Log log = LogFactory
43              .getLog(SessionSerializableCheckerValve.class);
44  
45      /**
46       * Check if all session attributes are serializable.
47       *
48       * @param request
49       *            the request being served
50       * @param response
51       *            the response being generated
52       * @throws IOException
53       * @throws ServletException
54       */
55      @Override
56      public final void invoke(final Request request, final Response response)
57              throws IOException, ServletException {
58          try {
59              getNext().invoke(request, response);
60          } finally {
61              if ((request.getSession(false) != null)) {
62                  @SuppressWarnings("unchecked")
63                  final Enumeration<String> attibuteNames = request.getSession()
64                          .getAttributeNames();
65                  while (attibuteNames.hasMoreElements()) {
66                      final String attributeName = attibuteNames.nextElement();
67                      final Object attribute = request.getSession().getAttribute(
68                              attributeName);
69                      try {
70                          checkSerializable(attribute);
71                      } catch (final Exception e) {
72                          log.warn(
73                                  format("Session attribute [%s] of class [%s] threw "
74                                          + "exception while serializing.",
75                                          attribute, attribute.getClass()), e);
76                      }
77                  }
78              }
79          }
80      }
81  
82      /**
83       * Check if an object is serializable, emit a warning log if it is not.
84       *
85       * @param attribute
86       *            the attribute to check
87       * @throws IOException
88       */
89      private void checkSerializable(final Object attribute) throws IOException {
90          if (!Serializable.class.isAssignableFrom(attribute.getClass())) {
91              log.warn(format("Session attribute [%s] of class [%s] is not "
92                      + "serializable.", attribute, attribute.getClass()));
93          }
94          final Closer closer = Closer.create();
95          try {
96              final ObjectOutputStream out = new ObjectOutputStream(
97                      ByteStreams.nullOutputStream());
98              out.writeObject(attribute);
99          } finally {
100             closer.close();
101         }
102     }
